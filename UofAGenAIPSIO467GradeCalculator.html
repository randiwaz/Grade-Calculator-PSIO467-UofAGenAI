<script>

(function() {

  'use strict';


  /* ── Course structure ─────────────────────────── */

  // Total possible points: 4 exams × 100 + 4 assignments × 25 = 500

  const EXAM_MAX   = 100;

  const ASSIGN_MAX = 25;

  const TOTAL_MAX  = 4 * EXAM_MAX + 4 * ASSIGN_MAX; // 500


  /* Grade thresholds (minimum % needed) */

  const THRESHOLDS = { A: 90.0, B: 80.0, C: 70.0, D: 60.0, E: 0 };


  let selectedGrade = 'A';


  /* ── Helpers ────────────────────────────────────── */

  function getVal(id) {

    const v = parseFloat(document.getElementById(id).value);

    return isNaN(v) ? null : v;

  }


  function fmt(n, dec) {

    if (n === null || n === undefined) return '—';

    return n.toFixed(dec !== undefined ? dec : 1);

  }


  function gradeFromPct(pct) {

    if (pct >= 90.0) return 'A';

    if (pct >= 80.0) return 'B';

    if (pct >= 70.0) return 'C';

    if (pct >= 60.0) return 'D';

    return 'E';

  }


  function barWidth(score, max) {

    if (score === null) return 0;

    const pct = Math.min((score / max) * 100, 115); // allow a little over for EC

    return Math.max(0, pct);

  }


  /* ── Input handler: style on entry ─────────────── */

  window.handleInput = function(input, type) {

    const max = (type === 'exam') ? EXAM_MAX : ASSIGN_MAX;

    if (input.value !== '' && !isNaN(parseFloat(input.value))) {

      input.classList.add('has-value');

    } else {

      input.classList.remove('has-value');

    }

    // Update bar

    const id = input.id;

    const barMap = {

      exam1:'bar-exam1', exam2:'bar-exam2', exam3:'bar-exam3',

      assign1:'bar-assign1', assign2:'bar-assign2', assign3:'bar-assign3', assign4:'bar-assign4'

    };

    const bar = document.getElementById(barMap[id]);

    if (bar) bar.style.width = barWidth(getVal(id), max) + '%';

    calculate();

  };


  /* ── Grade selector ─────────────────────────────── */

  window.selectGrade = function(grade) {

    selectedGrade = grade;

    document.querySelectorAll('.grade-btn').forEach(btn => {

      const isActive = btn.dataset.grade === grade;

      btn.classList.toggle('active', isActive);

      btn.setAttribute('aria-pressed', isActive ? 'true' : 'false');

    });

    calculate();

  };


  /* ── Highlight active grade in scale ───────────── */

  function highlightScale(letter) {

    ['A','B','C','D','E'].forEach(g => {

      const row = document.getElementById('scale-' + g);

      if (row) row.classList.toggle('active-grade', g === letter);

    });

  }


  /* ── Main calculate function ────────────────────── */

  window.calculate = function() {

    // Read values

    const e1 = getVal('exam1'), e2 = getVal('exam2'), e3 = getVal('exam3');

    const a1 = getVal('assign1'), a2 = getVal('assign2');

    const a3 = getVal('assign3'), a4 = getVal('assign4');


    /* ── Current Standing ── */

    const scores    = [e1, e2, e3, a1, a2, a3, a4];

    const maxPts    = [EXAM_MAX, EXAM_MAX, EXAM_MAX, ASSIGN_MAX, ASSIGN_MAX, ASSIGN_MAX, ASSIGN_MAX];

    let   earned    = 0;

    let   attempted = 0;


    scores.forEach((s, i) => {

      if (s !== null) {

        earned    += s;

        attempted += maxPts[i];

      }

    });


    const hasAny  = attempted > 0;

    const pct     = hasAny ? (earned / attempted) * 100 : null;

    const letter  = pct !== null ? gradeFromPct(pct) : null;


    // Update stat boxes

    const earnEl    = document.getElementById('stat-earned');

    const attEl     = document.getElementById('stat-attempted');

    const pctEl     = document.getElementById('stat-percent');

    const letEl     = document.getElementById('stat-letter');

    const letBox    = document.getElementById('stat-letter-box');


    earnEl.innerHTML    = hasAny ? `<span>${fmt(earned,1)}</span>` : `<span class="placeholder-dash">—</span>`;

    attEl.innerHTML     = hasAny ? `<span>${attempted}</span>` : `<span class="placeholder-dash">—</span>`;

    pctEl.innerHTML     = pct !== null ? `<span>${fmt(pct,1)}%</span>` : `<span class="placeholder-dash">—</span>`;

    letEl.innerHTML     = letter ? `<span>${letter}</span>` : `<span class="placeholder-dash">—</span>`;


    // Color letter box

    letBox.className = 'stat-box stat-letter' + (letter ? ' letter-' + letter : '');


    // Progress bar

    const fill  = document.getElementById('progress-fill');

    const lbl   = document.getElementById('progress-label');

    const pLbl  = document.getElementById('progress-pct-label');

    const pWrap = document.getElementById('progress-bar-wrap');


    if (pct !== null) {

      const clampedPct = Math.min(pct, 100);

      fill.style.width = clampedPct + '%';

      pLbl.textContent = fmt(pct, 1) + '%';

      lbl.textContent  = 'Overall performance';

      pWrap.setAttribute('aria-valuenow', Math.round(clampedPct));

      // Color the bar by grade

      const colors = { A:'#70B865', B:'#81D3EB', C:'#F4C842', D:'#F4A042', E:'#EF4056' };

      fill.style.background = colors[letter] || '#EF4056';

    } else {

      fill.style.width = '0%';

      pLbl.textContent = '0%';

      lbl.textContent  = 'Enter scores to begin';

      pWrap.setAttribute('aria-valuenow', 0);

      fill.style.background = 'linear-gradient(90deg, var(--az-red), var(--bloom))';

    }


    // Highlight scale

    highlightScale(letter);


    /* ── Target Grade Section ── */

    const allEntered = (e1 !== null && e2 !== null && e3 !== null &&

                        a1 !== null && a2 !== null && a3 !== null && a4 !== null);


    const hiddenEl  = document.getElementById('target-hidden');

    const contentEl = document.getElementById('target-content');


    if (!allEntered) {

      hiddenEl.classList.remove('hidden');

      hiddenEl.style.display = 'flex';

      contentEl.classList.remove('visible');

      contentEl.style.display = 'none';

      return;

    }


    // Show target section

    hiddenEl.classList.add('hidden');

    hiddenEl.style.display = 'none';

    contentEl.classList.add('visible');

    contentEl.style.display = 'block';


    computeTarget(e1, e2, e3, a1, a2, a3, a4);

  };


  /* ── Compute target exam 4 score ───────────────── */

  function computeTarget(e1, e2, e3, a1, a2, a3, a4) {

    const threshold = THRESHOLDS[selectedGrade];


    // Points earned so far (without exam 4)

    const soFar = e1 + e2 + e3 + a1 + a2 + a3 + a4;


    // Total possible with exam 4 included

    // total = soFar + exam4

    // final % = total / TOTAL_MAX * 100 >= threshold

    // exam4 >= (threshold/100 * TOTAL_MAX) - soFar


    const needed = (threshold / 100) * TOTAL_MAX - soFar;


    const mainTextEl  = document.getElementById('target-main-text');

    const eqEl        = document.getElementById('target-equation');

    const scoreBox    = document.getElementById('target-score-box');

    const scoreVal    = document.getElementById('target-score-value');

    const statusBar   = document.getElementById('target-status-bar');

    const statusIcon  = document.getElementById('target-status-icon');

    const statusText  = document.getElementById('target-status-text');


    // Reset classes

    scoreBox.className = 'target-score-box';

    statusBar.className = 'target-status-bar';


    const gradeLabel = { A:'an A', B:'a B', C:'a C', D:'a D', E:'an E (any score)' };


    if (selectedGrade === 'E') {

      // You can always fail — needs < 60%

      // Actually E means ≤ 59.9%, so technically you need to stay below that.

      // But the request is about "earning" any grade.

      // E: just need to show any score — even 0 may qualify.

      const maxPossible = soFar + EXAM_MAX;

      const maxPct = (maxPossible / TOTAL_MAX) * 100;


      if (maxPct < 60) {

        // Already guaranteed E

        mainTextEl.textContent = 'You will receive an E regardless of Exam 4.';

        scoreVal.textContent   = 'N/A';

        scoreBox.classList.add('tsb-guaranteed');

        eqEl.innerHTML = `<strong>Max possible:</strong> (${fmt(soFar,1)} + 100) / 500 = ${fmt(maxPct,1)}% &lt; 60%`;

        statusBar.classList.add('status-guaranteed');

        statusIcon.textContent = '📌';

        statusText.textContent = 'An E is mathematically guaranteed no matter what you score on Exam 4.';

      } else {

        // Need to score below a certain amount

        const maxNeeded = (60 / 100) * TOTAL_MAX - soFar - 0.01; // stay under 60%

        mainTextEl.textContent = `Score below ${fmt(maxNeeded,1)} pts on Exam 4 to earn an E.`;

        scoreVal.textContent   = fmt(maxNeeded, 1);

        scoreBox.classList.add('tsb-achievable');

        eqEl.innerHTML = `To stay below 60%: need Exam 4 &lt; <strong>${fmt(maxNeeded,1)}</strong>`;

        statusBar.classList.add('status-achievable');

        statusIcon.textContent = '⚠️';

        statusText.textContent = 'You can still avoid an E by scoring higher on Exam 4.';

      }

      return;

    }


    // Check if already guaranteed (even with 0 on exam 4)

    const minPossible = soFar + 0;

    const minPct = (minPossible / TOTAL_MAX) * 100;

    if (minPct >= threshold) {

      mainTextEl.textContent = `You've already secured ${gradeLabel[selectedGrade]}!`;

      scoreVal.textContent   = '0+';

      scoreBox.classList.add('tsb-guaranteed');

      eqEl.innerHTML = `Even with 0 on Exam 4: ${fmt(soFar,1)} / 500 = ${fmt(minPct,1)}% ≥ ${threshold}%`;

      statusBar.classList.add('status-guaranteed');

      statusIcon.textContent = '🎉';

      statusText.textContent = `${gradeLabel.charAt ? gradeLabel : gradeLabel[selectedGrade].charAt(0).toUpperCase() + gradeLabel[selectedGrade].slice(1)} is guaranteed — any Exam 4 score will do!`;

      statusText.textContent = `${gradeLabel[selectedGrade].charAt(0).toUpperCase() + gradeLabel[selectedGrade].slice(1)} is guaranteed — any Exam 4 score will do!`;

      return;

    }


    // Check if impossible (even with perfect score on exam 4)

    const maxPossible = soFar + EXAM_MAX;

    const maxPct = (maxPossible / TOTAL_MAX) * 100;

    if (maxPct < threshold) {

      mainTextEl.textContent = `It is no longer possible to earn ${gradeLabel[selectedGrade]}.`;

      scoreVal.textContent   = '> 100';

      scoreBox.classList.add('tsb-impossible');

      eqEl.innerHTML = `Max possible: (${fmt(soFar,1)} + 100) / 500 = ${fmt(maxPct,1)}% &lt; ${threshold}%`;

      statusBar.classList.add('status-impossible');

      statusIcon.textContent = '❌';

      statusText.textContent = `Even a perfect score on Exam 4 won't reach ${threshold}%. Consider a different target.`;

      return;

    }


    // Standard case: show needed score

    const neededRounded = Math.ceil(needed * 10) / 10; // round up to 1 decimal

    mainTextEl.textContent = `You need ${fmt(neededRounded, 1)} points on Exam 4 to earn ${gradeLabel[selectedGrade]}.`;

    scoreVal.textContent   = fmt(neededRounded, 1);

    scoreBox.classList.add('tsb-achievable');


    // Equation

    const finalPctIfPerfect = ((soFar + EXAM_MAX) / TOTAL_MAX * 100);

    eqEl.innerHTML =

      `(${fmt(soFar,1)} + <strong>x</strong>) / 500 ≥ ${threshold}% &nbsp;→&nbsp; <strong>x ≥ ${fmt(neededRounded,1)}</strong><br>` +

      `With 100 on Exam 4: ${fmt(finalPctIfPerfect,1)}% — max possible`;


    statusBar.classList.add('status-achievable');

    const pctNeeded = (neededRounded / EXAM_MAX) * 100;

    statusIcon.textContent = '✅';

    statusText.textContent = `You need ${fmt(pctNeeded,1)}% on Exam 4 (${fmt(neededRounded,1)}/100 points) to earn ${gradeLabel[selectedGrade]}.`;

  }


  /* ── Init bars on page load ─────────────────────── */

  (function initBars() {

    const inputs = ['exam1','exam2','exam3','assign1','assign2','assign3','assign4'];

    const types  = ['exam','exam','exam','assign','assign','assign','assign'];

    inputs.forEach((id, i) => {

      const el = document.getElementById(id);

      if (el && el.value !== '') handleInput(el, types[i]);

    });

  })();


})();
